# 扫雷 v2.3 核心可靠性实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 用受保护存档键和“最新快照”重试契约消除旧版覆盖/写盘失败丢进度，修复 0 秒成绩，并让无尽模式在所有随机候选下严格不倒退。

**Architecture:** `storage.ts` 保持 schema v3，但把 legacy 读取与受保护写入分离；内存 `SaveData` 是会话唯一真相，失败只保留一份最新 pending。`endless.ts` 用显式候选表替换事后抖动。`main.ts` 只消费存储返回值和 `flushPending()`，不得伪造 persisted。

**Tech Stack:** TypeScript、Vitest + jsdom、localStorage-compatible backend。

Task 1 阶段保留内部 `save`，但其中唯一 backend 写入必须精确为 `backend.setItem(SAVE_KEY, JSON.stringify(d))`，不得再出现 legacy key 写入。把返回对象中的 `recordWin` 方法完整替换为：

```text
recordWin(levelId, timeSec) {
  const d = {
    ...data,
    bestTimes: { ...data.bestTimes },
    endless: { ...data.endless },
  };
  const prev = d.bestTimes[levelId];
  const validTime = isValidTime(timeSec);
  const newBest = validTime && (prev === undefined || timeSec < prev);
  if (newBest) d.bestTimes[levelId] = timeSec;
  let unlocked: number | null = null;
  if (levelId < MAX_LEVEL && d.unlockedLevel === levelId) {
    d.unlockedLevel = levelId + 1;
    unlocked = d.unlockedLevel;
  }
  const persisted = save(d);
  return { newBest, unlocked, persisted };
},
```

因此非法时间不会产生伪成绩，而调用 `recordWin` 所代表的真实胜利仍按相同条件解锁下一关。

```ts
function switchableBackend(initial?: Record<string, string>) {
  const base = memBackend(initial);
  let failWrites = false;
  return {
    ...base,
    setFailWrites(fail: boolean) {
      failWrites = fail;
    },
    setItem(key: string, value: string) {
      if (failWrites) throw new Error("quota");
      base.map.set(key, value);
    },
  };
```

- [ ] **Step 4: 实现单一内存真相和最新 pending**

从 `export function createStorage` 签名到函数末尾完整替换，不能保留 Task 1 的旧 `let data` 或内部 `save`：

```ts
export function createStorage(backend?: Backend): GameStorage {
  let data = defaults();
  let pending: SaveData | null = null;

  const clone = (value: SaveData): SaveData => ({
    ...value,
    bestTimes: { ...value.bestTimes },
    endless: { ...value.endless },
  });

  const writeProtected = (value: SaveData): boolean => {
    if (!backend) return false;
    try {
      backend.setItem(SAVE_KEY, JSON.stringify(value));
      return true;
    } catch {
      return false;
    }
  };

  const commitLatest = (next: SaveData): boolean => {
    data = clone(next);
    const persisted = writeProtected(data);
    pending = persisted ? null : clone(data);
    return persisted;
  };

  const flushPending = (): FlushResult => {
    if (pending === null) return "idle";
    if (!writeProtected(pending)) return "failed";
    pending = null;
    return "saved";
  };

  let protectedMissing = false;
  try {
    const protectedRaw = backend?.getItem(SAVE_KEY);
    protectedMissing = protectedRaw == null;
    if (protectedRaw != null) data = sanitize(JSON.parse(protectedRaw) as unknown);
  } catch {
    protectedMissing = false;
    data = defaults();
  }

  if (protectedMissing) {
    try {
      const legacyRaw = backend?.getItem(LEGACY_SAVE_KEY);
      if (legacyRaw != null) {
        const migrated = sanitize(JSON.parse(legacyRaw) as unknown);
        commitLatest(migrated);
      }
    } catch {
      data = defaults();
    }
  }

  return {
    load: () => clone(data),
    flushPending,
    recordWin(levelId, timeSec) {
      const next = clone(data);
      const previous = next.bestTimes[levelId];
      const validTime = isValidTime(timeSec);
      const newBest = validTime && (previous === undefined || timeSec < previous);
      if (newBest) next.bestTimes[levelId] = timeSec;
      let unlocked: number | null = null;
      if (levelId < MAX_LEVEL && next.unlockedLevel === levelId) {
        next.unlockedLevel = levelId + 1;
        unlocked = next.unlockedLevel;
      }
      const persisted = commitLatest(next);
      return { newBest, unlocked, persisted };
    },
    setSoundOn(on) {
      const next = clone(data);
      next.soundOn = on;
      return commitLatest(next);
    },
    recordEndlessWin() {
      const next = clone(data);
      const streak = next.endless.streak + 1;
      const bestStreak = Math.max(next.endless.bestStreak, streak);
      const newBest = bestStreak > next.endless.bestStreak;
      next.endless = { streak, bestStreak };
      const persisted = commitLatest(next);
      return { streak, bestStreak, newBest, persisted };
    },
    recordEndlessLoss() {
      const next = clone(data);
      next.endless = { streak: 0, bestStreak: data.endless.bestStreak };
      const persisted = commitLatest(next);
      return { streak: 0, bestStreak: next.endless.bestStreak, persisted };
    },
  };
}
```

```ts
let persistenceWarning = backend === undefined;

function syncPersistenceWarning(): void {
  setPersistenceWarning(persistenceWarning);

function notePersisted(persisted: boolean): void {
  persistenceWarning = !persisted;

function retryPending(): void {
```

```ts
import type { LevelSpec } from "./levels";

export const ENDLESS_CAP_WINS = 15;
export const ENDLESS_MAX_DENSITY = 23.2;
export const ENDLESS_MIN_TIME = 900;
export const ENDLESS_MAX_TIME = 2400;

export interface EndlessShape {
  readonly width: number;
  readonly height: number;
}

export const ENDLESS_SHAPES_BY_STREAK = [
  [{ width: 20, height: 30 }, { width: 20, height: 31 }],
  [{ width: 21, height: 30 }, { width: 20, height: 32 }, { width: 21, height: 31 }],
  [{ width: 21, height: 32 }, { width: 22, height: 31 }, { width: 21, height: 33 }],
  [{ width: 22, height: 32 }, { width: 22, height: 33 }, { width: 22, height: 34 }],
  [{ width: 23, height: 33 }, { width: 22, height: 35 }, { width: 23, height: 34 }],
  [{ width: 23, height: 35 }, { width: 24, height: 34 }, { width: 23, height: 36 }],
  [{ width: 24, height: 35 }, { width: 24, height: 36 }, { width: 25, height: 35 }],
  [{ width: 24, height: 37 }, { width: 25, height: 36 }, { width: 24, height: 38 }],
  [{ width: 25, height: 37 }, { width: 25, height: 38 }, { width: 26, height: 37 }],
  [{ width: 25, height: 39 }, { width: 26, height: 38 }, { width: 25, height: 40 }],
  [{ width: 26, height: 39 }, { width: 27, height: 38 }, { width: 26, height: 40 }],
  [{ width: 27, height: 39 }, { width: 26, height: 41 }, { width: 27, height: 40 }],
  [{ width: 27, height: 41 }, { width: 28, height: 40 }, { width: 27, height: 42 }],
  [{ width: 28, height: 41 }, { width: 27, height: 43 }, { width: 28, height: 42 }],
  [{ width: 28, height: 43 }, { width: 28, height: 44 }],
] as const satisfies readonly (readonly EndlessShape[])[];

function densityAt(streak: number): number {
  return 22.1 + ((ENDLESS_MAX_DENSITY - 22.1) * Math.min(streak, ENDLESS_CAP_WINS)) /
    ENDLESS_CAP_WINS;
}

export function endlessSpec(streak: number, rng: () => number): LevelSpec {
  const n = Number.isFinite(streak) ? Math.max(0, Math.floor(streak)) : 0;
  if (n >= ENDLESS_CAP_WINS) {
    return { id: 0, tier: "endless", width: 28, height: 44, mines: 285, timeLimitSec: 2400 };
  }
  const shapes = ENDLESS_SHAPES_BY_STREAK[n]!;
  const rawIndex = Math.floor(Math.max(0, rng()) * shapes.length);
  const shape = shapes[Math.min(shapes.length - 1, rawIndex)]!;
  const cells = shape.width * shape.height;
  return {
    id: 0,
    tier: "endless",
    width: shape.width,
    height: shape.height,
    mines: Math.floor((cells * densityAt(n)) / 100),
    timeLimitSec: Math.min(ENDLESS_MIN_TIME + n * 100, ENDLESS_MAX_TIME),
  };
}
```
