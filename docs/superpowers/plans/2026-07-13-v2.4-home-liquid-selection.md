```ts
export type HomeLiquidTargetKind = "instant" | "navigation";

export interface HomeLiquidTarget {
  button: HTMLButtonElement;
  kind: HomeLiquidTargetKind;
  activate(): void;
}

export interface HomeLiquidSelectionController {
  destroy(): void;
}

interface Box {
  x: number;
  y: number;
  width: number;
  height: number;
}

const NAVIGATION_DELAY_MS = 220;
const CLICK_DURATION_MS = 580;
const TARGET_PADDING_PX = 5;
const MIN_TARGET_PX = 48;

const clamp = (value: number, min: number, max: number): number =>
  Math.max(min, Math.min(max, value));

export function installHomeLiquidSelection(
  panel: HTMLElement,
  indicator: HTMLElement,
  targets: readonly HomeLiquidTarget[],
  initialButton: HTMLButtonElement,
): HomeLiquidSelectionController {
  const ownerWindow = panel.ownerDocument.defaultView;
  if (!ownerWindow) throw new Error("home liquid selection requires a window");
  const byButton = new Map(targets.map((target) => [target.button, target]));
  const initial = byButton.get(initialButton);
  if (!initial || initial.button.disabled) throw new Error("invalid initial home liquid target");

  let current = initial;
  let animation: Animation | null = null;
  let pendingActivation = 0;
  let destroyed = false;

  const measure = (button: HTMLButtonElement): Box => {
    const panelRect = panel.getBoundingClientRect();
    const targetRect = button.getBoundingClientRect();
    const rawWidth = Math.max(MIN_TARGET_PX, targetRect.width + TARGET_PADDING_PX * 2);
    const rawHeight = Math.max(MIN_TARGET_PX, targetRect.height + TARGET_PADDING_PX * 2);
    const width = panelRect.width > 12 ? Math.min(rawWidth, panelRect.width - 12) : rawWidth;
    const height = panelRect.height > 12 ? Math.min(rawHeight, panelRect.height - 12) : rawHeight;
    const naturalX = targetRect.left + targetRect.width / 2 - panelRect.left;
    const naturalY = targetRect.top + targetRect.height / 2 - panelRect.top;
    return {
      x: panelRect.width > 0 ? clamp(naturalX, width / 2 + 6, panelRect.width - width / 2 - 6) : naturalX,
      y: panelRect.height > 0 ? clamp(naturalY, height / 2 + 6, panelRect.height - height / 2 - 6) : naturalY,
      width,
      height,
    };
  };

  const styleFor = (box: Box, sx = 1, sy = 1): Keyframe => ({
    left: `${box.x}px`,
    top: `${box.y}px`,
    width: `${box.width}px`,
    height: `${box.height}px`,
    transform: `translate(-50%, -50%) scale(${sx}, ${sy})`,
  });

  const apply = (box: Box): void => {
    Object.assign(indicator.style, styleFor(box));
    indicator.style.opacity = "1";
  };

  const mark = (target: HomeLiquidTarget): void => {
    for (const item of targets) {
      item.button.classList.toggle("is-home-selected", item === target);
    }
    current = target;
  };

  const reducedMotion = (): boolean =>
    ownerWindow.matchMedia?.("(prefers-reduced-motion: reduce)").matches ?? false;

  const animateTo = (target: HomeLiquidTarget): boolean => {
    const changed = target !== current;
    const from = measure(current.button);
    const to = measure(target.button);
    mark(target);
    animation?.cancel();
    animation = null;
    if (!changed || reducedMotion() || typeof indicator.animate !== "function") {
      apply(to);
      return changed;
    }
    const middle: Box = {
      x: (from.x + to.x) / 2,
      y: (from.y + to.y) / 2,
      width: (from.width + to.width) / 2,
      height: (from.height + to.height) / 2,
    };
    animation = indicator.animate([
      { ...styleFor(from), offset: 0 },
      { ...styleFor(middle, 1.22, 0.88), offset: 0.52 },
      { ...styleFor(to, 1.08, 0.94), offset: 0.78 },
      { ...styleFor(to, 0.96, 1.05), offset: 0.9 },
      { ...styleFor(to), offset: 1 },
    ], { duration: CLICK_DURATION_MS, easing: "cubic-bezier(.16,1,.3,1)", fill: "forwards" });
    animation.onfinish = () => {
      animation = null;
      apply(to);
    };
    return changed;
  };

  const clearPendingActivation = (): void => {
    if (pendingActivation !== 0) ownerWindow.clearTimeout(pendingActivation);
    pendingActivation = 0;
  };

  const activate = (target: HomeLiquidTarget, changed: boolean): void => {
    clearPendingActivation();
    const run = (): void => {
      pendingActivation = 0;
      if (!destroyed && target.button.isConnected) target.activate();
    };
    if (target.kind === "navigation" && changed) {
      pendingActivation = ownerWindow.setTimeout(run, NAVIGATION_DELAY_MS);
    } else {
      run();
    }
  };

  const onClick = (event: Event): void => {
    const button = event.target instanceof Element
      ? event.target.closest<HTMLButtonElement>("button")
      : null;
    const target = button ? byButton.get(button) : undefined;
    if (!target || target.button.disabled) return;
    const changed = animateTo(target);
    activate(target, changed);
  };

  const onResize = (): void => {
    animation?.cancel();
    animation = null;
    apply(measure(current.button));
  };

  panel.addEventListener("click", onClick);
  ownerWindow.addEventListener("resize", onResize);
  mark(initial);
  apply(measure(initial.button));

  return {
    destroy(): void {
      if (destroyed) return;
      destroyed = true;
      clearPendingActivation();
      animation?.cancel();
      animation = null;
      panel.removeEventListener("click", onClick);
      ownerWindow.removeEventListener("resize", onResize);
      for (const target of targets) {
        target.button.classList.remove("is-home-selected", "is-home-candidate");
      }
    },
  };
}
```

```ts
  type: string,
  target: EventTarget,
  init: { id: number; x: number; y: number; pointerType?: string; button?: number },
): void {
  const event = new MouseEvent(type, {
    bubbles: true,
    cancelable: true,
    clientX: init.x,
    clientY: init.y,
    button: init.button ?? 0,
  });
  Object.defineProperties(event, {
    pointerId: { value: init.id },
    pointerType: { value: init.pointerType ?? "touch" },
    isPrimary: { value: true },
  });
  target.dispatchEvent(event);
}
```

```ts
const clearCandidates = (): void => {
  for (const target of targets) target.button.classList.remove("is-home-candidate");
};

const available = (): HomeLiquidTarget[] => targets.filter((target) => !target.button.disabled);

const nearest = (point: Point): { target: HomeLiquidTarget; box: Box; distance: number } | null => {
  let best: { target: HomeLiquidTarget; box: Box; distance: number } | null = null;
  for (const target of available()) {
    const box = measure(target.button);
    const distance = Math.hypot(point.x - box.x, point.y - box.y);
    if (!best || distance < best.distance) best = { target, box, distance };
  }
  if (!best) return null;
  const threshold = Math.max(64, Math.min(105, Math.hypot(best.box.width, best.box.height) * 0.58));
  return best.distance < threshold ? best : null;
};

const panelPoint = (clientX: number, clientY: number): Point => {
  const bounds = panel.getBoundingClientRect();
  return { x: clientX - bounds.left, y: clientY - bounds.top };
};

const writeOptics = (clientX: number, clientY: number): void => {
  const bounds = panel.getBoundingClientRect();
  const x = bounds.width > 0 ? clamp((clientX - bounds.left) / bounds.width, 0, 1) : 0.5;
  const y = bounds.height > 0 ? clamp((clientY - bounds.top) / bounds.height, 0, 1) : 0.5;
  panel.style.setProperty("--glass-x", `${Math.round(x * 10000) / 100}%`);
  panel.style.setProperty("--glass-y", `${Math.round(y * 10000) / 100}%`);
  panel.style.setProperty("--glass-dx", String((x - 0.5) * 2));
  panel.style.setProperty("--glass-dy", String((y - 0.5) * 2));
};

const animateBack = (): void => {
  const to = measure(current.button);
  animation?.cancel();
  animation = null;
  if (reducedMotion() || typeof indicator.animate !== "function") {
    apply(to);
    return;
  }
  const from = indicator.getBoundingClientRect();
  const panelBounds = panel.getBoundingClientRect();
  const start: Box = {
    x: from.left + from.width / 2 - panelBounds.left,
    y: from.top + from.height / 2 - panelBounds.top,
    width: from.width,
    height: from.height,
  };
  animation = indicator.animate([
    { ...styleFor(start), offset: 0 },
    { ...styleFor(to, 1.04, 0.96), offset: 0.72 },
    { ...styleFor(to), offset: 1 },
  ], { duration: SETTLE_DURATION_MS, easing: "cubic-bezier(.16,1,.3,1)", fill: "forwards" });
  animation.onfinish = () => { animation = null; apply(to); };
};

const removeActiveListeners = (): void => {
  ownerWindow.removeEventListener("pointermove", onPointerMove, true);
  ownerWindow.removeEventListener("pointerup", onPointerUp, true);
  ownerWindow.removeEventListener("pointercancel", onPointerCancel, true);
  ownerWindow.removeEventListener("blur", onWindowBlur, true);
  panel.ownerDocument.removeEventListener("visibilitychange", onVisibilityChange, true);
};

const releaseCapture = (pointerId: number): void => {
  try {
    if (panel.hasPointerCapture?.(pointerId)) panel.releasePointerCapture(pointerId);
  } catch {
    // Capture can already be gone on Safari/iOS interruption paths.
  }
};

const clearMoveFrame = (): void => {
  if (moveFrame !== null) cancelFrame(moveFrame);
  moveFrame = null;
  pendingMove = null;
};

const cancelDrag = (animate: boolean): void => {
  if (!drag) return;
  const pointerId = drag.id;
  drag = null;
  clearMoveFrame();
  removeActiveListeners();
  releaseCapture(pointerId);
  clearCandidates();
  panel.classList.remove("is-home-liquid-dragging");
  if (!destroyed) animate ? animateBack() : apply(measure(current.button));
};

const applyMove = (sample: PointerSample): void => {
  if (!drag || drag.id !== sample.id) return;
  const elapsed = Math.max(8, sample.time - drag.lastTime);
  const velocityX = (sample.x - drag.last.x) / elapsed;
  const velocityY = (sample.y - drag.last.y) / elapsed;
  if (Math.hypot(sample.x - drag.start.x, sample.y - drag.start.y) > 5) drag.moved = true;
  const candidate = nearest(sample);
  const base = candidate?.box ?? drag.base;
  const proximity = candidate ? Math.max(0, 1 - candidate.distance / 110) : 0;
  const width = base.width * (candidate?.target === current ? 1 : 0.78 + proximity * 0.22);
  const height = base.height * (0.9 + proximity * 0.1);
  const bounds = panel.getBoundingClientRect();
  const x = clamp(sample.x, width / 2 + 6, bounds.width - width / 2 - 6);
  const y = clamp(sample.y, height / 2 + 6, bounds.height - height / 2 - 6);
  const sx = 1 + Math.min(0.28, Math.abs(velocityX) * 0.24)
    - Math.min(0.1, Math.abs(velocityY) * 0.06);
  const sy = 1 + Math.min(0.2, Math.abs(velocityY) * 0.2)
    - Math.min(0.1, Math.abs(velocityX) * 0.05);
  Object.assign(indicator.style, styleFor({ x, y, width, height }, sx, sy));
  clearCandidates();
  if (candidate && candidate.target !== current) {
    candidate.target.button.classList.add("is-home-candidate");
  }
  drag.candidate = candidate?.target ?? null;
  drag.last = { x: sample.x, y: sample.y };
  drag.lastTime = sample.time;
};

const flushMove = (): void => {
  if (moveFrame !== null) cancelFrame(moveFrame);
  moveFrame = null;
  const sample = pendingMove;
  pendingMove = null;
  if (sample) applyMove(sample);
};

function onPointerMove(event: PointerEvent): void {
  if (!drag || event.pointerId !== drag.id) return;
  event.preventDefault();
  const point = panelPoint(event.clientX, event.clientY);
  pendingMove = { id: event.pointerId, ...point, time: ownerWindow.performance.now() };
  writeOptics(event.clientX, event.clientY);
  if (moveFrame === null) moveFrame = requestFrame(flushMove);
}

const finishDrag = (event: PointerEvent, cancelled: boolean): void => {
  if (!drag || event.pointerId !== drag.id) return;
  flushMove();
  const state = drag;
  drag = null;
  removeActiveListeners();
  releaseCapture(state.id);
  clearCandidates();
  panel.classList.remove("is-home-liquid-dragging");
  if (state.moved) suppressClickUntil = ownerWindow.performance.now() + SETTLE_DURATION_MS;
  if (cancelled || !state.moved || !state.candidate || state.candidate === current) {
    animateBack();
    return;
  }
  const target = state.candidate;
  const changed = animateTo(target);
  activate(target, changed);
};

function onPointerUp(event: PointerEvent): void { finishDrag(event, false); }
function onPointerCancel(event: PointerEvent): void { finishDrag(event, true); }
function onWindowBlur(): void { cancelDrag(true); }
function onVisibilityChange(): void {
  if (panel.ownerDocument.visibilityState === "hidden") cancelDrag(true);
}

const onPointerDown = (event: PointerEvent): void => {
  if (destroyed || event.isPrimary === false) return;
  if (event.pointerType === "mouse" && event.button !== 0) return;
  const button = event.target instanceof Element
    ? event.target.closest<HTMLButtonElement>("button")
    : null;
  const target = button ? byButton.get(button) : undefined;
  const grabbedIndicator = event.target === indicator || indicator.contains(event.target as Node);
  if (!grabbedIndicator && target !== current) return;
  if (drag) cancelDrag(false);
  animation?.cancel();
  animation = null;
  apply(measure(current.button));
  const start = panelPoint(event.clientX, event.clientY);
  drag = {
    id: event.pointerId,
    start,
    last: start,
    lastTime: ownerWindow.performance.now(),
    base: measure(current.button),
    moved: false,
    candidate: null,
  };
  panel.classList.add("is-home-liquid-dragging");
  ownerWindow.addEventListener("pointermove", onPointerMove, { capture: true, passive: false });
  ownerWindow.addEventListener("pointerup", onPointerUp, true);
  ownerWindow.addEventListener("pointercancel", onPointerCancel, true);
  ownerWindow.addEventListener("blur", onWindowBlur, true);
  panel.ownerDocument.addEventListener("visibilitychange", onVisibilityChange, true);
  try { panel.setPointerCapture?.(event.pointerId); } catch {
    // Window listeners are the complete fallback when capture is unavailable.
  }
};

const onLostPointerCapture = (event: PointerEvent): void => {
  if (drag && event.pointerId === drag.id) cancelDrag(true);
};
```

